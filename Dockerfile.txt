# Use Python 3.11 slim image
FROM python:3.11-slim

# Install system packages (ffmpeg and dependencies)
RUN apt-get update && apt-get install -y ffmpeg gcc libffi-dev libssl-dev && rm -rf /var/lib/apt/lists/*

# Set working directory
WORKDIR /app

# Install Python dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application code
COPY . .

# Start API server using Gunicorn (Railway sets PORT env automatically)
CMD gunicorn --bind 0.0.0.0:${PORT:-8080} --reuse-port main:app
