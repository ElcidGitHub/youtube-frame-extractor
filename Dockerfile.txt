# Use lightweight Python 3.11 image
FROM python:3.11-slim

# Set environment variables to reduce layer size and avoid Python bytecode
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1

# Install required OS packages
RUN apt-get update && apt-get install -y \
    ffmpeg \
    yt-dlp \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Set working directory
WORKDIR /app

# Copy dependency file first for caching
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy rest of the app code
COPY . .

# Expose port (for local Docker run; Railway uses PORT env automatically)
EXPOSE 5000

# Start app with Gunicorn and auto Railway port
CMD gunicorn --bind 0.0.0.0:${PORT:-5000} --reuse-port main:app
